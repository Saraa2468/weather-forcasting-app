pipeline {
    agent any

    stages {
        stage('Build Docker Image') {
            steps {
                script {
                    sh 'docker build -t weather-app:jenkins .'
                }
            }
        }

        stage('Run Tests') {
            steps {
                script {
                    sh 'python test.py'
                }
            }
        }

        stage('Deploy to Kubernetes') {
            steps {
                script {
                    sh 'helm upgrade --install weather-app ./weather-chart'
                }
            }
        }
    }
}
